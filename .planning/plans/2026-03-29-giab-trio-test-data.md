# GIAB Chinese Trio Test Data Preparation — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Produce a small, committed set of VCF test files from the GIAB Chinese Trio for VCF import and trio analysis feature development.

**Architecture:** Two bash scripts — a one-time setup script (`setup-varlens-tools.sh`) that installs the conda env + Docker image + caches, and a data preparation script (`prepare-test-data.sh`) that downloads GIAB data, subsets, merges, cherry-picks edge cases, annotates with VEP and SnpEff, and packages final files. A synthetic VCF is generated inline with no external dependencies.

**Tech Stack:** bash, bcftools, htslib (bgzip/tabix), Docker (VEP), conda/mamba (SnpEff/SnpSift), GIAB benchmark VCFs (GRCh38 v4.2.1)

**Parallelization:** Tasks 1-3 are independent and can run in parallel. Task 4 depends on Task 1 (setup must exist). Tasks 5-7 depend on Task 4 (need conda env + tools). Tasks within the prepare script are sequential by nature (download → subset → merge → annotate), but VEP and SnpEff annotation are independent of each other and run in parallel inside the script.

## Task 1: Create the Setup Script (Independent — can run in parallel with Tasks 2, 3)

**Files:**
- Create: `scripts/setup-varlens-tools.sh`

This script installs all prerequisites. It is run once manually before `prepare-test-data.sh`.

- [ ] **Step 1: Create setup script with header and configuration**

```bash
cat > scripts/setup-varlens-tools.sh << 'SCRIPT_EOF'
#!/usr/bin/env bash
# =============================================================================
# setup-varlens-tools.sh — One-time setup for VarLens test data tools
#
# Installs:
#   - conda env 'varlens-tools' (bcftools, htslib, snpeff, snpsift)
#   - Docker image ensemblorg/ensembl-vep:release_115.2
#   - VEP GRCh38 cache (~20 GB)
#   - SnpEff GRCh38 database
#   - ClinVar GRCh38 VCF
#
# Usage:
#   ./scripts/setup-varlens-tools.sh [OPTIONS]
#
# Options:
#   --skip-conda      Skip conda env creation
#   --skip-docker     Skip Docker image pull
#   --skip-vep-cache  Skip VEP cache download (largest component, ~20 GB)
#   --skip-clinvar    Skip ClinVar VCF download
#   --help            Show this help
#
# Prerequisites:
#   - conda or mamba in PATH
#   - Docker installed and running
#   - ~25 GB free disk space (VEP cache + SnpEff DB + ClinVar)
# =============================================================================
set -euo pipefail

# === Configuration ===
CONDA_ENV="varlens-tools"
VEP_IMAGE="ensemblorg/ensembl-vep:release_115.2"
VEP_CACHE_DIR="${HOME}/.vep"
VEP_RELEASE="115"
VEP_ASSEMBLY="GRCh38"
SNPEFF_DB="GRCh38.mane.1.2.ensembl"
CLINVAR_URL="https://ftp.ncbi.nlm.nih.gov/pub/clinvar/vcf_GRCh38/clinvar.vcf.gz"
CLINVAR_DIR="${HOME}/data/varlens-testdata/downloads"

# === Flags ===
SKIP_CONDA=false
SKIP_DOCKER=false
SKIP_VEP_CACHE=false
SKIP_CLINVAR=false

# === Logging ===
log() { echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*"; }
log_success() { log "✓ $*"; }
log_skip() { log "⊘ SKIP: $*"; }
log_error() { log "✗ ERROR: $*" >&2; }

# === Argument parsing ===
while [[ $# -gt 0 ]]; do
  case "$1" in
    --skip-conda)    SKIP_CONDA=true; shift ;;
    --skip-docker)   SKIP_DOCKER=true; shift ;;
    --skip-vep-cache) SKIP_VEP_CACHE=true; shift ;;
    --skip-clinvar)  SKIP_CLINVAR=true; shift ;;
    --help)
      sed -n '2,/^# =====/{ /^# =====/d; s/^# //; s/^#//; p }' "$0"
      exit 0
      ;;
    *) log_error "Unknown option: $1"; exit 1 ;;
  esac
done

# === Detect conda/mamba ===
if command -v mamba &>/dev/null; then
  CONDA_CMD="mamba"
elif command -v conda &>/dev/null; then
  CONDA_CMD="conda"
else
  log_error "Neither mamba nor conda found in PATH"
  exit 1
fi
log "Using ${CONDA_CMD} for environment management"

# === Step 1: Conda environment ===
if [[ "$SKIP_CONDA" == true ]]; then
  log_skip "conda env creation"
elif conda env list 2>/dev/null | grep -q "^${CONDA_ENV} "; then
  log_skip "conda env '${CONDA_ENV}' already exists"
else
  log "Creating conda env '${CONDA_ENV}' with bcftools, htslib, snpeff, snpsift..."
  ${CONDA_CMD} create -y -n "${CONDA_ENV}" -c bioconda -c conda-forge \
    bcftools htslib snpeff snpsift
  log_success "conda env '${CONDA_ENV}' created"
fi

# === Step 2: Docker VEP image ===
if [[ "$SKIP_DOCKER" == true ]]; then
  log_skip "Docker image pull"
elif docker image inspect "${VEP_IMAGE}" &>/dev/null; then
  log_skip "Docker image '${VEP_IMAGE}' already exists"
else
  log "Pulling Docker image '${VEP_IMAGE}'..."
  docker pull "${VEP_IMAGE}"
  log_success "Docker image pulled"
fi

# === Step 3: VEP cache ===
if [[ "$SKIP_VEP_CACHE" == true ]]; then
  log_skip "VEP cache download"
elif [[ -d "${VEP_CACHE_DIR}/homo_sapiens/${VEP_RELEASE}_${VEP_ASSEMBLY}" ]]; then
  log_skip "VEP cache already exists at ${VEP_CACHE_DIR}/homo_sapiens/${VEP_RELEASE}_${VEP_ASSEMBLY}"
else
  log "Downloading VEP ${VEP_ASSEMBLY} cache (release ${VEP_RELEASE})..."
  log "This is ~20 GB and will take a while."
  mkdir -p "${VEP_CACHE_DIR}"
  CACHE_URL="https://ftp.ensembl.org/pub/release-${VEP_RELEASE}/variation/indexed_vep_cache/homo_sapiens_vep_${VEP_RELEASE}_${VEP_ASSEMBLY}.tar.gz"
  wget -c -P "${VEP_CACHE_DIR}" "${CACHE_URL}"
  log "Extracting cache..."
  tar xzf "${VEP_CACHE_DIR}/homo_sapiens_vep_${VEP_RELEASE}_${VEP_ASSEMBLY}.tar.gz" -C "${VEP_CACHE_DIR}"
  rm -f "${VEP_CACHE_DIR}/homo_sapiens_vep_${VEP_RELEASE}_${VEP_ASSEMBLY}.tar.gz"
  log_success "VEP cache installed at ${VEP_CACHE_DIR}/homo_sapiens/${VEP_RELEASE}_${VEP_ASSEMBLY}"
fi

# === Step 4: SnpEff database ===
log "Checking SnpEff database '${SNPEFF_DB}'..."
# Activate conda env temporarily for snpEff
eval "$(conda shell.bash hook)"
conda activate "${CONDA_ENV}" 2>/dev/null || true

if snpEff databases 2>/dev/null | grep -q "${SNPEFF_DB}"; then
  # Check if already downloaded by trying a dry-run
  if snpEff dump "${SNPEFF_DB}" &>/dev/null; then
    log_skip "SnpEff database '${SNPEFF_DB}' already downloaded"
  else
    log "Downloading SnpEff database '${SNPEFF_DB}'..."
    snpEff download "${SNPEFF_DB}"
    log_success "SnpEff database downloaded"
  fi
else
  log_error "SnpEff database '${SNPEFF_DB}' not found in available databases"
  log "Trying download anyway (may auto-resolve)..."
  snpEff download "${SNPEFF_DB}" || log_error "Failed to download SnpEff database"
fi

conda deactivate 2>/dev/null || true

# === Step 5: ClinVar VCF ===
if [[ "$SKIP_CLINVAR" == true ]]; then
  log_skip "ClinVar download"
elif [[ -f "${CLINVAR_DIR}/clinvar.vcf.gz" && -f "${CLINVAR_DIR}/clinvar.vcf.gz.tbi" ]]; then
  log_skip "ClinVar VCF already exists at ${CLINVAR_DIR}/clinvar.vcf.gz"
else
  log "Downloading ClinVar GRCh38 VCF..."
  mkdir -p "${CLINVAR_DIR}"
  wget -c -P "${CLINVAR_DIR}" "${CLINVAR_URL}"
  wget -c -P "${CLINVAR_DIR}" "${CLINVAR_URL}.tbi"
  log_success "ClinVar VCF downloaded to ${CLINVAR_DIR}/"
fi

# === Verify ===
log ""
log "=== Setup Summary ==="
log "Conda env:    $(conda env list 2>/dev/null | grep "^${CONDA_ENV} " && echo 'OK' || echo 'MISSING')"
log "Docker VEP:   $(docker image inspect "${VEP_IMAGE}" &>/dev/null && echo 'OK' || echo 'MISSING')"
log "VEP cache:    $([[ -d "${VEP_CACHE_DIR}/homo_sapiens/${VEP_RELEASE}_${VEP_ASSEMBLY}" ]] && echo 'OK' || echo 'MISSING')"
log "ClinVar:      $([[ -f "${CLINVAR_DIR}/clinvar.vcf.gz" ]] && echo 'OK' || echo 'MISSING')"
log ""
log "Setup complete. Run ./scripts/prepare-test-data.sh to generate test data."
SCRIPT_EOF
chmod +x scripts/setup-varlens-tools.sh
```

- [ ] **Step 2: Verify script is syntactically valid**

Run: `bash -n scripts/setup-varlens-tools.sh`
Expected: No output (no syntax errors)

```bash

check_prereqs() {
  log "Checking prerequisites..."
  local ok=true

  # conda env
  if ! conda env list 2>/dev/null | grep -q "^${CONDA_ENV} "; then
    log_err "Conda env '${CONDA_ENV}' not found. Run ./scripts/setup-varlens-tools.sh first."
    ok=false
  fi

  # Docker + VEP image
  if ! command -v docker &>/dev/null; then
    log_err "Docker not found in PATH."
    ok=false
  elif ! docker image inspect "${VEP_IMAGE}" &>/dev/null; then
    log_err "Docker image '${VEP_IMAGE}' not found. Run ./scripts/setup-varlens-tools.sh first."
    ok=false
  fi

  # VEP cache
  if [[ ! -d "${VEP_CACHE_DIR}/homo_sapiens" ]]; then
    log_err "VEP cache not found at ${VEP_CACHE_DIR}/homo_sapiens/. Run ./scripts/setup-varlens-tools.sh first."
    ok=false
  fi

  # ClinVar (needed for SnpSift)
  if [[ ! -f "${CLINVAR_VCF}" ]]; then
    log_warn "ClinVar VCF not found at ${CLINVAR_VCF}. SnpSift ClinVar annotation will be skipped."
  fi

  # Disk space (need at least 2 GB for intermediates)
  local free_gb
  free_gb=$(df -BG "${WORKDIR%/*}" 2>/dev/null | tail -1 | awk '{print $4}' | sed 's/G//')
  if [[ -n "$free_gb" && "$free_gb" -lt 2 ]]; then
    log_err "Less than 2 GB free in ${WORKDIR%/*}. Need space for downloads and intermediates."
    ok=false
  fi

  if [[ "$ok" != true ]]; then
    log_err "Prerequisite check failed. Aborting."
    exit 1
  fi
  log_ok "All prerequisites met"
}

# =============================================================================
# Step 1: Download GIAB benchmark VCFs
# =============================================================================

step_download() {
  log_step "Download GIAB benchmark VCFs"
  mkdir -p "${DOWNLOAD_DIR}"

  for i in "${!SAMPLES[@]}"; do
    local sample="${SAMPLES[$i]}"
    local full_name="${SAMPLE_NAMES[$i]}"
    local vcf_name="${sample}_${BUILD}_1_22_${BENCHMARK_VERSION}_benchmark.vcf.gz"
    local vcf_url="${BASE_URL}/${full_name}/latest/${BUILD}/${vcf_name}"

    # Download VCF
    if should_skip "${DOWNLOAD_DIR}/${vcf_name}"; then
      log_skip "${vcf_name} (already downloaded)"
    else
      log "Downloading ${vcf_name}..."
      wget -c -q --show-progress -P "${DOWNLOAD_DIR}" "${vcf_url}"
      log_ok "Downloaded ${vcf_name}"
    fi

    # Download index
    if should_skip "${DOWNLOAD_DIR}/${vcf_name}.tbi"; then
      log_skip "${vcf_name}.tbi (already downloaded)"
    else
      log "Downloading ${vcf_name}.tbi..."
      wget -c -q --show-progress -P "${DOWNLOAD_DIR}" "${vcf_url}.tbi"
      log_ok "Downloaded ${vcf_name}.tbi"
    fi
  done
}

# =============================================================================
# Step 2: Subset to chr22 per sample
# =============================================================================

step_subset() {
  log_step "Subset to chr22 per sample"
  mkdir -p "${INTERMEDIATE_DIR}"
  activate_conda

  for sample in "${SAMPLES[@]}"; do
    local input="${DOWNLOAD_DIR}/${sample}_${BUILD}_1_22_${BENCHMARK_VERSION}_benchmark.vcf.gz"
    local output="${INTERMEDIATE_DIR}/${sample}_${BUILD}_chr22.vcf.gz"

    if should_skip "${output}"; then
      log_skip "${sample} chr22 subset (already exists)"
      continue
    fi

    if [[ ! -f "$input" ]]; then
      log_err "Input not found: ${input}. Run --step download first."
      exit 1
    fi

    log "Extracting chr22 from ${sample}..."
    bcftools view -r chr22 "${input}" -Oz -o "${output}"
    bcftools index -t "${output}"

    # Fix sample name if needed (GIAB may use different naming)
    local current_name
    current_name=$(bcftools query -l "${output}")
    if [[ "$current_name" != "$sample" ]]; then
      log "Renaming sample '${current_name}' → '${sample}'..."
      echo "${current_name} ${sample}" > "${INTERMEDIATE_DIR}/rename_${sample}.txt"
      bcftools reheader -s "${INTERMEDIATE_DIR}/rename_${sample}.txt" "${output}" \
        -o "${output}.tmp"
      mv "${output}.tmp" "${output}"
      bcftools index -t "${output}"
      rm -f "${INTERMEDIATE_DIR}/rename_${sample}.txt"
    fi

    local count
    count=$(bcftools stats "${output}" | grep "^SN.*number of records:" | awk '{print $NF}')
    log_ok "${sample} chr22: ${count} variants"
  done
}

# Step 3: Merge into multi-sample trio VCF + extract single sample

step_merge() {

  activate_conda

  local trio_vcf="${INTERMEDIATE_DIR}/chinese_trio_${BUILD}_chr22.vcf.gz"

  if should_skip "${trio_vcf}"; then
    log_skip "Trio VCF (already exists)"

    bcftools index -t "${trio_vcf}"

    log "Samples in merged VCF:"
    bcftools query -l "${trio_vcf}"

    local count
    count=$(bcftools stats "${trio_vcf}" | grep "^SN.*number of records:" | awk '{print $NF}')
    log_ok "Trio chr22 merged: ${count} variants, 3 samples"
  fi

  # Extract region subset for the final test file
  local region_vcf="${INTERMEDIATE_DIR}/trio_region.vcf.gz"

  if should_skip "${region_vcf}"; then
    log_skip "Region subset (already exists)"
  else
    log "Extracting region ${REGION}..."
    bcftools view -r "${REGION}" "${trio_vcf}" -Oz -o "${region_vcf}"
    bcftools index -t "${region_vcf}"

    local count
    count=$(bcftools stats "${region_vcf}" | grep "^SN.*number of records:" | awk '{print $NF}')
    log_ok "Region subset: ${count} variants"
  fi

  local single_vcf="${INTERMEDIATE_DIR}/single_sample.vcf.gz"

}

# =============================================================================
# Step 4: Cherry-pick edge case variants
# =============================================================================

step_cherry_pick() {
  log_step "Cherry-pick edge case variants"
  activate_conda

  local trio_vcf="${INTERMEDIATE_DIR}/chinese_trio_${BUILD}_chr22.vcf.gz"
  local positions_file="${INTERMEDIATE_DIR}/edge_case_positions.txt"
  local edge_vcf="${INTERMEDIATE_DIR}/edge_cases.vcf.gz"

  if should_skip "${edge_vcf}"; then
    log_skip "Edge cases VCF (already exists)"
    return
  fi

  if [[ ! -f "$trio_vcf" ]]; then
    log_err "Trio VCF not found. Run --step merge first."
    exit 1
  fi

  # Collect positions for each edge case category
  > "${positions_file}"  # truncate

  log "Finding de novo candidates (son=0/1, both parents=0/0)..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'GT[0]="0/1" && GT[1]="0/0" && GT[2]="0/0"' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true
  local de_novo_count
  de_novo_count=$(bcftools query -f '%CHROM\t%POS\n' \
    -i 'GT[0]="0/1" && GT[1]="0/0" && GT[2]="0/0"' \
    "${trio_vcf}" | wc -l)
  log "  Found ${de_novo_count} total, picked ${CHERRY_PICK_COUNT}"

  log "Finding hom recessive candidates (son=1/1, both parents=0/1)..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'GT[0]="1/1" && GT[1]="0/1" && GT[2]="0/1"' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true

  log "Finding autosomal dominant candidates (son=0/1, one parent=0/1, other=0/0)..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'GT[0]="0/1" && ((GT[1]="0/1" && GT[2]="0/0") || (GT[1]="0/0" && GT[2]="0/1"))' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true

  log "Finding multi-allelic sites..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'N_ALT>1' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true

  log "Finding missing genotypes..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'GT[0]="mis" || GT[1]="mis" || GT[2]="mis"' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true

  log "Finding indels..."
  bcftools query -f '%CHROM\t%POS\n' \
    -i 'TYPE="indel"' \
    "${trio_vcf}" | head -${CHERRY_PICK_COUNT} >> "${positions_file}" || true

  # Deduplicate positions and convert to regions format
  sort -k1,1 -k2,2n "${positions_file}" | uniq > "${positions_file}.dedup"
  mv "${positions_file}.dedup" "${positions_file}"

  local total_positions
  total_positions=$(wc -l < "${positions_file}")
  log "Total unique edge case positions: ${total_positions}"

  if [[ "$total_positions" -eq 0 ]]; then
    log_warn "No edge case variants found! Creating empty edge-cases VCF with header only."
    bcftools view -h "${trio_vcf}" | bgzip > "${edge_vcf}"
    bcftools index -t "${edge_vcf}"
    return
  fi

  # Extract those positions from the trio VCF
  bcftools view -R "${positions_file}" "${trio_vcf}" -Oz -o "${edge_vcf}"
  bcftools index -t "${edge_vcf}"

  local count
  count=$(bcftools stats "${edge_vcf}" | grep "^SN.*number of records:" | awk '{print $NF}')
  log_ok "Edge cases: ${count} variants from ${total_positions} positions"
}

# =============================================================================
# Step 5: Annotate with VEP (Docker)
# =============================================================================

step_annotate_vep() {
  log_step "Annotate with VEP (Docker)"

  local input="${INTERMEDIATE_DIR}/trio_region.vcf.gz"
  local output="${INTERMEDIATE_DIR}/trio_region.vep.vcf"
  local output_gz="${INTERMEDIATE_DIR}/trio_region.vep.vcf.gz"

  if should_skip "${output_gz}"; then
    log_skip "VEP-annotated VCF (already exists)"
    return
  fi

  if [[ ! -f "$input" ]]; then
    log_err "Region VCF not found at ${input}. Run --step merge first."
    exit 1
  fi

  log "Running VEP annotation via Docker..."
  log "Input: ${input}"
  log "VEP cache: ${VEP_CACHE_DIR}"

  docker run --rm \
    -v "${INTERMEDIATE_DIR}:/data" \
    -v "${VEP_CACHE_DIR}:/opt/vep/.vep" \
    "${VEP_IMAGE}" \
    vep \
      --input_file /data/trio_region.vcf.gz \
      --output_file /data/trio_region.vep.vcf \
      --vcf \
      --cache --offline \
      --assembly "${BUILD}" \
      --everything \
      --pick \
      --fork 4 \
      --force_overwrite \
      --no_stats

  log "Compressing and indexing VEP output..."
  activate_conda
  bgzip -f "${output}"
  tabix -p vcf "${output_gz}"

  local count
  count=$(bcftools stats "${output_gz}" | grep "^SN.*number of records:" | awk '{print $NF}')
  log_ok "VEP annotation complete: ${count} variants"

  # Verify CSQ header is present
  if bcftools view -h "${output_gz}" | grep -q "^##INFO=<ID=CSQ"; then
    log_ok "CSQ INFO header confirmed"
  else
    log_err "CSQ INFO header NOT found in VEP output!"
    exit 1
  fi
}

# =============================================================================
# Step 6: Annotate with SnpEff + SnpSift (conda)
# =============================================================================

step_annotate_snpeff() {
  log_step "Annotate with SnpEff + SnpSift (conda)"
  activate_conda

  local input="${INTERMEDIATE_DIR}/trio_region.vcf.gz"
  local snpeff_out="${INTERMEDIATE_DIR}/trio_region.snpeff.vcf"
  local output_gz="${INTERMEDIATE_DIR}/trio_region.snpeff.vcf.gz"

  if should_skip "${output_gz}"; then
    log_skip "SnpEff-annotated VCF (already exists)"
    return
  fi

  if [[ ! -f "$input" ]]; then
    log_err "Region VCF not found at ${input}. Run --step merge first."
    exit 1
  fi

  log "Running SnpEff functional annotation..."

  if [[ -f "${CLINVAR_VCF}" ]]; then
    # Pipe: SnpEff → SnpSift ClinVar
    log "  SnpEff ann → SnpSift annotate (ClinVar)"
    snpEff ann -v -noStats "${SNPEFF_DB}" "${input}" 2>"${LOG_DIR}/snpeff.log" \
      | SnpSift annotate -name CLINVAR_ \
          -info CLNSIG,CLNDN,CLNREVSTAT \
          "${CLINVAR_VCF}" \
          2>"${LOG_DIR}/snpsift.log" \
      > "${snpeff_out}"
  else
    # SnpEff only (no ClinVar)
    log_warn "ClinVar VCF not found — running SnpEff without ClinVar."
    snpEff ann -v -noStats "${SNPEFF_DB}" "${input}" \
      2>"${LOG_DIR}/snpeff.log" \
      > "${snpeff_out}"
  fi

  log "Compressing and indexing SnpEff output..."
  bgzip -f "${snpeff_out}"
  tabix -p vcf "${output_gz}"

  local count
  count=$(bcftools stats "${output_gz}" | grep "^SN.*number of records:" | awk '{print $NF}')
  log_ok "SnpEff annotation complete: ${count} variants"

  # Verify ANN header is present
  if bcftools view -h "${output_gz}" | grep -q "^##INFO=<ID=ANN"; then
    log_ok "ANN INFO header confirmed"
  else
    log_err "ANN INFO header NOT found in SnpEff output!"
    exit 1
  fi
}

# Step 7: Package final files into output directory

step_package() {
  log_step "Package final files to ${OUTDIR}"
  activate_conda
  mkdir -p "${OUTDIR}"

  # Copy VCF files

    "trio_region.vcf.gz:trio-region.vcf.gz"
    "trio_region.vcf.gz.tbi:trio-region.vcf.gz.tbi"
    "trio_region.vep.vcf.gz:trio-region.vep.vcf.gz"
    "trio_region.vep.vcf.gz.tbi:trio-region.vep.vcf.gz.tbi"
    "trio_region.snpeff.vcf.gz:trio-region.snpeff.vcf.gz"

    "single_sample.vcf.gz.tbi:single-sample.vcf.gz.tbi"
    "edge_cases.vcf.gz:edge-cases.vcf.gz"
    "edge_cases.vcf.gz.tbi:edge-cases.vcf.gz.tbi"
  )

  for pair in "${files_to_copy[@]}"; do
    local src="${INTERMEDIATE_DIR}/${pair%%:*}"
    local dst="${OUTDIR}/${pair##*:}"
    if [[ -f "$src" ]]; then
      cp "$src" "$dst"
      log_ok "Copied ${pair##*:}"
    else
      log_warn "Source not found: ${src} — skipping"
    fi
  done

  # Generate PED file (matches sample order in VCF: HG005, HG006, HG007)
  log "Generating PED file..."
  cat > "${OUTDIR}/chinese_trio.ped" << 'PEDEOF'
#Family_ID	Individual_ID	Paternal_ID	Maternal_ID	Sex	Phenotype
CHINESE_TRIO	HG005	HG006	HG007	1	2
CHINESE_TRIO	HG006	0	0	1	1
CHINESE_TRIO	HG007	0	0	2	1
PEDEOF
  log_ok "PED file generated"

  # Generate README.md with provenance
  log "Generating README.md..."
  {
    echo "# VCF Test Data — GIAB Chinese Trio"
    echo ""
    echo "Auto-generated by \`scripts/prepare-test-data.sh\` on $(date '+%Y-%m-%d %H:%M:%S')."
    echo "**DO NOT EDIT MANUALLY** — regenerate with \`./scripts/prepare-test-data.sh --force\`."
    echo ""
    echo "## Source"
    echo ""
    echo "- **GIAB Chinese Trio** (HG005=son/proband, HG006=father, HG007=mother)"
    echo "- **Genome build:** GRCh38"
    echo "- **Benchmark version:** ${BENCHMARK_VERSION}"
    echo "- **Region:** ${REGION}"
    echo ""
    echo "## Tool Versions"
    echo ""
    echo "| Tool | Version |"
    echo "|------|---------|"
    echo "| bcftools | $(bcftools --version | head -1) |"

    # VEP version from Docker
    local vep_version
    vep_version=$(docker run --rm "${VEP_IMAGE}" vep --help 2>&1 | grep "Versions:" -A5 | head -3 || echo "unknown")
    echo "| VEP | ${VEP_IMAGE} |"

    echo "| snpEff | $(snpEff -version 2>&1 | head -1 || echo 'unknown') |"
    echo "| SnpSift | $(SnpSift -version 2>&1 | head -1 || echo 'unknown') |"
    echo ""
    echo "## Files"
    echo ""
    echo "| File | Variants | Samples | Size | md5sum |"
    echo "|------|----------|---------|------|--------|"

    for f in "${OUTDIR}"/*.vcf.gz; do
      if [[ -f "$f" ]]; then
        local fname
        fname=$(basename "$f")
        local variants
        variants=$(bcftools stats "$f" 2>/dev/null | grep "^SN.*number of records:" | awk '{print $NF}' || echo "?")
        local samples
        samples=$(bcftools query -l "$f" 2>/dev/null | wc -l || echo "?")
        local size
        size=$(du -h "$f" | cut -f1)
        local md5
        md5=$(md5sum "$f" | cut -d' ' -f1)
        echo "| ${fname} | ${variants} | ${samples} | ${size} | \`${md5}\` |"
      fi
    done

    # Non-VCF files
    for f in "${OUTDIR}/synthetic-unit-test.vcf" "${OUTDIR}/chinese_trio.ped"; do
      if [[ -f "$f" ]]; then
        local fname
        fname=$(basename "$f")
        local size
        size=$(du -h "$f" | cut -f1)
        local md5
        md5=$(md5sum "$f" | cut -d' ' -f1)
        echo "| ${fname} | — | — | ${size} | \`${md5}\` |"
      fi
    done

    echo ""
    echo "## Reproduction"
    echo ""
    echo "\`\`\`bash"
    echo "# One-time setup (installs tools, ~25 GB for caches)"
    echo "./scripts/setup-varlens-tools.sh"
    echo ""
    echo "# Generate test data"
    echo "./scripts/prepare-test-data.sh --step all"
    echo "\`\`\`"
  } > "${OUTDIR}/README.md"
  log_ok "README.md generated"
}

step_verify() {
  log_step "Verify outputs"
  activate_conda

  local all_ok=true

  echo ""
  echo "File Verification:"
  echo "─────────────────────────────────────────────────────────"

  local expected_files=(
    "trio-region.vcf.gz"
    "trio-region.vcf.gz.tbi"
    "trio-region.vep.vcf.gz"
    "trio-region.vep.vcf.gz.tbi"
    "trio-region.snpeff.vcf.gz"
    "trio-region.snpeff.vcf.gz.tbi"
    "single-sample.vcf.gz"
    "single-sample.vcf.gz.tbi"
    "edge-cases.vcf.gz"
    "edge-cases.vcf.gz.tbi"
    "chinese_trio.ped"
    "synthetic-unit-test.vcf"
    "README.md"
  )

  for f in "${expected_files[@]}"; do
    if [[ -f "${OUTDIR}/${f}" ]]; then
      local size
      size=$(du -h "${OUTDIR}/${f}" | cut -f1)
      printf "  %-40s %8s  %s\n" "$f" "$size" "OK"
    else
      printf "  %-40s %8s  %s\n" "$f" "—" "MISSING"
      all_ok=false
    fi
  done

  # Check total size
  echo ""
  local total_size
  total_size=$(du -sh "${OUTDIR}" | cut -f1)
  echo "Total size: ${total_size}"

  # Verify annotations in annotated files
  echo ""
  echo "Annotation Verification:"
  echo "─────────────────────────────────────────────────────────"

  if [[ -f "${OUTDIR}/trio-region.vep.vcf.gz" ]]; then
    if bcftools view -h "${OUTDIR}/trio-region.vep.vcf.gz" | grep -q "^##INFO=<ID=CSQ"; then
      echo "  trio-region.vep.vcf.gz    CSQ header: PRESENT"
    else
      echo "  trio-region.vep.vcf.gz    CSQ header: MISSING"
      all_ok=false
    fi
  fi

  if [[ -f "${OUTDIR}/trio-region.snpeff.vcf.gz" ]]; then
    if bcftools view -h "${OUTDIR}/trio-region.snpeff.vcf.gz" | grep -q "^##INFO=<ID=ANN"; then
      echo "  trio-region.snpeff.vcf.gz ANN header: PRESENT"
    else
      echo "  trio-region.snpeff.vcf.gz ANN header: MISSING"
      all_ok=false
    fi
  fi

  # Verify sample names match PED
  if [[ -f "${OUTDIR}/trio-region.vcf.gz" ]]; then
    local vcf_samples
    vcf_samples=$(bcftools query -l "${OUTDIR}/trio-region.vcf.gz" | sort | tr '\n' ',')
    echo ""
    echo "  VCF samples: ${vcf_samples}"
    echo "  PED samples: HG005,HG006,HG007,"
    if [[ "$vcf_samples" == "HG005,HG006,HG007," ]]; then
      echo "  Sample match: OK"
    else
      echo "  Sample match: MISMATCH"
      all_ok=false
    fi
  fi

  echo ""
  if [[ "$all_ok" == true ]]; then
    log_ok "All verifications passed!"
  else
    log_err "Some verifications failed. Check output above."
    exit 1
  fi
}

main() {
  log "VarLens Test Data Preparation"
  log "Working directory: ${WORKDIR}"
  log "Output directory:  ${OUTDIR}"
  log "Region:            ${REGION}"
  log "Steps:             ${STEPS}"
  log "Force:             ${FORCE}"
  log ""

  mkdir -p "${DOWNLOAD_DIR}" "${INTERMEDIATE_DIR}" "${LOG_DIR}"

  if [[ "$DRY_RUN" == true ]]; then
    log "DRY RUN — would execute these steps:"
    for step in download subset merge cherry-pick annotate-vep annotate-snpeff package verify; do
      if should_run_step "$step"; then
        echo "  - ${step}"
      fi
    done
    exit 0
  fi

  check_prereqs

  should_run_step "download"       && step_download
  should_run_step "subset"         && step_subset
  should_run_step "merge"          && step_merge
  should_run_step "cherry-pick"    && step_cherry_pick

  # VEP and SnpEff annotation can run in parallel
  if should_run_step "annotate-vep" && should_run_step "annotate-snpeff"; then
    log "Running VEP and SnpEff annotations in parallel..."
    step_annotate_vep &
    local vep_pid=$!
    step_annotate_snpeff &
    local snpeff_pid=$!

    local fail=false
    wait $vep_pid || { log_err "VEP annotation failed"; fail=true; }
    wait $snpeff_pid || { log_err "SnpEff annotation failed"; fail=true; }
    [[ "$fail" == true ]] && exit 1
  else
    should_run_step "annotate-vep"    && step_annotate_vep
    should_run_step "annotate-snpeff" && step_annotate_snpeff
  fi

  should_run_step "package"        && step_package
  should_run_step "verify"         && step_verify

  if [[ "$CLEANUP" == true ]]; then
    log "Cleaning up intermediate files..."
    rm -rf "${INTERMEDIATE_DIR}"
    log_ok "Intermediates removed"
  fi

  log ""
  log_ok "Done! Test data ready at ${OUTDIR}"
}
```
